package com.example;

import java.io.IOException;
import java.sql.*;
import java.util.Random;

public class VulnerableApp {

    // Hardcoded credentials (S2068)
    private static final String PASSWORD = "SuperSecret123!";

    public static void main(String[] args) {
        VulnerableApp app = new VulnerableApp();
        app.sqlInjection("101 OR 1=1");
        app.commandInjection("hello; rm -rf /");
        app.weakRandom();
        app.insecureHashing();
    }

    // SQL Injection (S3649, S2077)
    public void sqlInjection(String input) {
        try {
            Connection conn = DriverManager.getConnection("jdbc:h2:mem:test", "sa", PASSWORD);
            Statement stmt = conn.createStatement();

            String query = "SELECT * FROM users WHERE id = '" + input + "'";
            System.out.println("Executing query: " + query);

            ResultSet rs = stmt.executeQuery(query);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    // Command Injection (S2076)
    public void commandInjection(String input) {
        try {
            Runtime.getRuntime().exec("echo " + input);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    // Weak random number generation (S2245)
    public void weakRandom() {
        Random r = new Random();
        System.out.println("Weak Token: " + r.nextInt());
    }

    // Insecure hashing (S4790)
    public void insecureHashing() {
        try {
            java.security.MessageDigest md = java.security.MessageDigest.getInstance("MD5"); // Weak hashing
            md.update("password".getBytes());
            System.out.println("MD5 Hash: " + new String(md.digest()));
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
